CREATE TABLE subtitle (
	subtitleID INT NOT NULL primary key, /* 자막 ID */
	subtitleFileURL VARCHAR2(255), /* 자막 파일 URL */
	lang VARCHAR2(30), /* 자막 */
	videoID INT NOT NULL, /* 영상 ID */
	subtitleSettingID INT NOT NULL, /* 자막표시 ID */
    CONSTRAINT FK_video_TO_subtitle FOREIGN KEY (videoID) REFERENCES video(videoID),
    CONSTRAINT FK_subtitleSetting_TO_subtitle FOREIGN KEY (subtitleSettingID) REFERENCES subtitleSetting(subtitleSettingID)
);

각 video별로 서로 다른 언어의 자막을 정의하는 테이블이다.

1. subtitle테이블의 데이터는 위에 작성한 video 테이블에 정의된 데이터들을 기반으로 작성된다.
2. pk는 시퀀스로 1부터 순차적으로 증가한다.
3. 모든 데이터는 넷플릭스 사이트의 정보를 기반으로 작성되어야 한다.
4. lang은 자막에 정의된 언어를 의미하며 한국어의 경우 한국어, 영어인 경우 영어 이런식으로 작성된다.
5. video테이블에 정의된 모든 데이터에 대해 최소 한 개 이상씩 존재해야하며 해당 video의 기본 언어를 기반으로 작성되어야 한다. 예를 들면 기생충의 경우 videoid는 1 lang은 한국어 이렇게 작성되어야 한다.
6. 넷플릭스에서 video의 각각의 데이터에 대해 기본언어 외 자막을 제공한다면 그것을 다른 데이터로 분류한다.
7. 테이블 명과 컬럼명을 변경하지 않는다.

위 인덱스의 요구사항을 준수한다. 데이터를 출력해야 되는 테이블은 배우와 배우리스트 테이블들이다. 저장은 csv파일 형식으로 저장할 것이며 이는 복사 붙혀넣기를 통해 저장할 것이다.
